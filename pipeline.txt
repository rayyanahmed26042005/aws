pipeline {
    agent any

    tools {
        maven 'MAVEN_HOME'
    }

    stages {

        stage('git repo & clean') {
            steps {
                // Safe delete: avoid failure if SE folder doesn't exist
                bat '''
                IF EXIST SE (
                    echo "Deleting old SE directory..."
                    rmdir /s /q SE
                ) ELSE (
                    echo "SE directory not found, skipping delete."
                )
                '''

                // Clone the repo again
                bat "git clone https://github.com/rayyanahmed26042005/SE.git"

                // Clean the project
                bat "mvn -f SE/pom.xml clean"
            }
        }

        stage('install') {
            steps {
                bat "mvn -f SE/pom.xml install"
            }
        }

        stage('test') {
            steps {
                bat "mvn -f SE/pom.xml test"
            }
        }

        stage('package') {
            steps {
                bat "mvn -f SE/pom.xml package"
            }
        }
    }
}
